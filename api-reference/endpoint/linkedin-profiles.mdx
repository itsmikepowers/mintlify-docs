        required: true

          application/json:
            schema:
              type: 'object'
              properties:
                links:
                  type: 'array'

                    - 'http://www.linkedin.com/in/rodneydbainjr'

                  type: 'boolean'
                  description: 'Option to flatten the response structure'

              required:
                - links
                - flatten
      responses:
        '200':
          description: 'Successful response with LinkedIn profile details'
          content:
            application/json:
              schema:
                type: 'array'
                items:
                  type: 'object'
                  additionalProperties: true

            application/json:
              schema:
                $ref: '#/components/schemas/Error'
