---
title: 'Linkedin Companies'
openapi: 'POST /maps/linkedin-companies'
description: 'Search Linkedin to get company data.'
parameters:
  - name: 'links'
    in: 'body'
    description: 'List of LinkedIn company URLs'
    required: true
    schema:
      type: 'array'
      items:
        type: 'string'
      example:
        - "http://www.linkedin.com/company/aep-energy"
        - "http://www.linkedin.com/company/johnson-&-johnson"
        - "http://www.linkedin.com/company/ametek"
  - name: 'flatten'
    in: 'body'
    description: 'Option to flatten the response structure'
    required: false
    schema:
      type: 'boolean'
      example: false
responses:
  '200':
    description: 'Successful response with LinkedIn company details'
    content:
      application/json:
        schema:
          type: 'array'
          items:
            type: 'object'
            additionalProperties: true
  '400':
    description: 'Unexpected error'
    content:
      application/json:
        schema:
          $ref: '#/components/schemas/Error'
components:
  schemas:
    Error:
      type: 'object'
      properties:
        error:
          type: 'integer'
          format: 'int32'
        message:
          type: 'string'
      required:
        - error
        - message
---
